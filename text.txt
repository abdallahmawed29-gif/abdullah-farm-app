  <script>
    async function loadCrops() {
      try {
        const response = await fetch("crops.json");
        const crops = await response.json();

        const container = document.getElementById("cropsContainer");
        container.innerHTML = "";

        crops.forEach(crop => {
          const div = document.createElement("div");
          div.className = "crop";
          div.textContent = crop.name;
          div.onclick = () => {
            alert(`تم اختيار: ${crop.name}`);
          };
          container.appendChild(div);
        });
      } catch (error) {
        console.error("خطأ في تحميل المحاصيل:", error);
      }
    }

    loadCrops();
  </script>
